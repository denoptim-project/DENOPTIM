87 public ArrayList<IAtomContainer> buildMulti3DStructure()
88 throws DENOPTIMException
89 {
90 logger.log(Level.INFO, "Building Multiple 3D representations for "
91 + "graph = " + molGraph.toString());
92
93 // Generate XYZ and INT representations
94 long startTime = System.nanoTime();
95 ChemicalObjectModel mol = build3DTree();
96 long endTime = System.nanoTime();
97 long time = (endTime - startTime);
98
99 logger.log(Level.FINE, "TIME (build 3D model): " + time/1000000 + " ms"
100 + " #frags: " + mol.getGraph().getVertexList().size()
101 + " #atoms: " + mol.getIAtomContainer().getAtomCount());
102
103 // get settings //TODO: this should happen inside RunTimeParameters
104 RingClosureParameters rcParams = new RingClosureParameters();
105 if (settings.containsParameters(ParametersType.RC_PARAMS))
106 {
107 rcParams = (RingClosureParameters)settings.getParameters(
108 ParametersType.RC_PARAMS);
109 }
110
111 // Evaluate source of isomerism
112 // 1: Attempt Ring Closures
113 RingClosureTool rct = new RingClosureTool(settings);
114 ArrayList<ChemicalObjectModel> structures =new ArrayList<ChemicalObjectModel>();
115 boolean skipConfSearch = false;
116 if (rcParams.allowRingClosures() && mol.getGraph().hasOrEmbedsRings())
117 {
118 startTime = System.nanoTime();
119 structures = rct.attemptAllRingClosures(mol);
120 endTime = System.nanoTime();
121 time = (endTime - startTime);
122 int numAllClosedCombs = 0;
123 for (ChemicalObjectModel rcMol : structures)
124 {
125 Object o = rcMol.getIAtomContainer().getProperty(
126 DENOPTIMConstants.MOLERRORTAG);
127 if (o == null)
128 numAllClosedCombs++;
129 }
130 logger.log(Level.FINE, "TIME (close ring): "+time/1000000+" ms"
131 + " #frags: " + mol.getGraph().getVertexList().size()
132 + " #atoms: " + mol.getIAtomContainer().getAtomCount()
133 + " #rcaCombs: " + mol.getRCACombinations().size()
134 + " #allClosedRCSCombs: " + numAllClosedCombs);
135 if (rcParams.requireCompleteRingclosure && numAllClosedCombs<1)
136 {
137 logger.log(Level.INFO, "No fully closed RCA combinaton. "
138 + "Nothing to send to conformational search.");
139 skipConfSearch = true;
140 }
141 // Ring-closing conf search on the RCO server includes general
142 // purpose conformational search, so no need to run a second
143 // conformational search.
144 // RCO server is the default, so the condition checks for non-default
145 // configuration.
146 if (settings.getPSSROTTool() == null)
147 {
148 skipConfSearch = true;
149 }
150 } else {
151 ChemicalObjectModel nMol = mol.deepcopy();
152 rct.saturateRingClosingAttractor(nMol);
153 structures = new ArrayList<ChemicalObjectModel>();
154 structures.add(nMol);
155 }
156
157 // 2: Conformational search (if possible)
158 if (!skipConfSearch)
159 {
160 if (settings.getPSSROTTool() != null)
161 {
162 try
163 {
164 startTime = System.nanoTime();
165 ConformationalSearchPSSROT.performPSSROT(structures,
166 settings.getTinkerMap(),
167 "cs",
168 settings.getParamFile(),
169 settings.getKeyFileParams(),
170 settings.getInitPSSROTParams(),
171 settings.getRestPSSROTParams(),
172 settings.getPSSROTTool(),
173 settings.getXYZINTTool(),
174 settings.getWorkingDirectory(),
175 settings.getTaskID(), logger);
176 time = (endTime - startTime);
177 } catch (TinkerException te)
178 {
179 String msg = "ERROR! Tinker failed on task '"
180 + te.taskName + "'!";
181 if (te.solution != "")
182 {
183 msg = msg + settings.NL + te.solution;
184 }
185 logger.log(Level.SEVERE, msg);
186 throw new DENOPTIMException(msg, te);
187 }
188 } else {
189 RCOSocketServerClient rcoServer = RCOSocketServerClient.getInstance(
190 settings.getRCOServerHostname(), settings.getRCOServerPort());
191 for (ChemicalObjectModel com : structures)
192 {
193 try
194 {
195 rcoServer.runConformationalOptimization(com, logger);
196 } catch (Exception e) {
197 e.printStackTrace();
198 String msg = "ERROR! RCOSocketServer failed!";
199 logger.log(Level.SEVERE, msg);
200 throw new DENOPTIMException(msg, e);
201 }
202
203 }
204 }
205 logger.log(Level.FINE, "TIME (conf. search): "+time/1000000+" ms"
206 + " #frags: " + mol.getGraph().getVertexList().size()
207 + " #atoms: " + mol.getIAtomContainer().getAtomCount()
208 + " #rotBnds: " + mol.getRotatableBonds().size());
209 }
210
211 // Convert and return results
212 ArrayList<IAtomContainer> results = new ArrayList<IAtomContainer>();
213 DummyAtomHandler dah = new DummyAtomHandler(
214 DENOPTIMConstants.DUMMYATMSYMBOL, logger);
215 for (ChemicalObjectModel mol3db : structures)
216 {
217 IAtomContainer iac = mol3db.getIAtomContainer();
218
219 IAtomContainer originalOrderMol = AtomOrganizer.makeReorderedCopy(
220 iac, mol3db.getOldToNewOrder(), mol3db.getNewToOldOrder());
221 iac = originalOrderMol;
222
223 if (!settings.getKeepDummyFlag())
224 {
225 // To keep track of which vertexID should be removed from mol
226 // properties, we remove that property from the mol. It remains
227 // defined in each atom.
228 iac.removeProperty(DENOPTIMConstants.ATMPROPVERTEXID);
229
230 iac = dah.removeDummyInHapto(iac);
231 iac = dah.removeDummy(iac);
232
233 // Now we put the property back among the molecular ones
234 StringBuilder sbMolProp = new StringBuilder();
235 for (IAtom atm : iac.atoms())
236 {
237 sbMolProp.append(" ").append(atm.getProperty(
238 DENOPTIMConstants.ATMPROPVERTEXID).toString());
239 }
240 iac.setProperty(DENOPTIMConstants.ATMPROPVERTEXID,
241 sbMolProp.toString().trim());
242 results.add(iac);
243 } else {
244 results.add(iac);
245 }
246 }
247
248 return results;
249 }

261 public ChemicalObjectModel build3DTree() throws DENOPTIMException
262 {
263 // Create 3D tree-like structure
264 ThreeDimTreeBuilder tb = new ThreeDimTreeBuilder(logger,
265 settings.getRandomizer());
266 IAtomContainer initMol = tb.convertGraphTo3DAtomContainer(molGraph,
267 false, //don't remove used RCAs
268 true, //set the CDK requirements
269 true); //rebuild low-quality structured embedded in templates
270
271 // Reorder atoms and clone molecule.
272 AtomOrganizer oa = new AtomOrganizer();
273 oa.setScheme(settings.getAtomOrderingScheme());
274 int seedAtm = 0;
275 IAtomContainer reorderedMol = oa.reorderStartingFrom(seedAtm, initMol);
276 ArrayList<Integer> newToOldMap = oa.getNewToOldOrder(seedAtm);
277 ArrayList<Integer> oldToNewMap = oa.getOldToNewOrder(seedAtm);
278 logger.log(Level.FINEST, "oldToNewMap: "+oldToNewMap);
279
280 // Collect rotatable bonds defined by fragment-fragment connections
281 ArrayList<ObjectPair> rotBonds = RotationalSpaceUtils
282 .defineRotatableBonds(reorderedMol,
283 ((FragmentSpaceParameters) settings.getParameters(
284 ParametersType.FS_PARAMS)).getRotSpaceDefFile(),
285 true, true, settings.getLogger());
286
287 //Apply constraints, if any
288 RotationalSpaceUtils.processConstrainedRotatableBonds(reorderedMol,
289 rotBonds, ((FragmentSpaceParameters) settings.getParameters(
290 ParametersType.FS_PARAMS)).getRotConstraintDefFile(), logger);
291
292 logger.log(Level.FINE, "Rotatable bonds: "+rotBonds);
293 if (logger.isLoggable(Level.FINEST))
294 {
295 logger.log(Level.FINEST, "Reordered IAtomContainer: 'iacToIC.sdf'");
296 DenoptimIO.writeSDFFile("iacToIC.sdf",reorderedMol,false);
297 }
298
299 // Generate Internal Coordinates
300 ZMatrix zmat= ZMatrix.getZMatrixFromIAC(reorderedMol);
301
302 // Generate combined molecular representations (both XYZ and INT)
303 return new ChemicalObjectModel(
304 molGraph,
305 reorderedMol,
306 zmat,
307 molName,
308 rotBonds,
309 oldToNewMap,
310 newToOldMap,
311 logger
312 );
313 }
