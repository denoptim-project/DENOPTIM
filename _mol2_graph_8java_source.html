65 @Override
66 public void runProgram() throws Throwable
67 {
68 Mol2GraphParameters m2gParams = new Mol2GraphParameters();
69 m2gParams.readParameterFile(configFilePathName.getAbsolutePath());
70 m2gParams.checkParameters();
71 m2gParams.processParameters();
72 m2gParams.startProgramSpecificLogger(loggerIdentifier, false); //to STDOUT
73
74 List<DGraph> graphs = new ArrayList<DGraph>();
75 for (int i=0; i<m2gParams.getInputMolsCount(); i++)
76 {
77 IAtomContainer mol = m2gParams.getInputMol(i);
78 DGraph graph = null;
79 try {
80 graph = EAUtils.makeGraphFromFragmentationOfMol(mol,
81 m2gParams.getCuttingRules(),
82 m2gParams.getLogger(),
83 m2gParams.getScaffoldingPolicy(),
84 m2gParams.getLinearAngleLimit(),
85 m2gParams.embedRingsInTemplate(),
86 m2gParams.getEmbeddedRingsContract(),
87 m2gParams.getFragmentSpace(),
88 null); // monitor is not used here
89 } catch (DENOPTIMException de)
90 {
91 m2gParams.getLogger().log(Level.SEVERE, "Unable to convert "
92 + "molecule " + i + " to DENOPTIM graph. "
93 + de.getMessage());
94 return;
95 }
96
97 graphs.add(graph);
98 }
99
100 if (graphs.size()>1)
101 {
102 DenoptimIO.writeGraphsToFile(new File(m2gParams.getOutFile()),
103 m2gParams.getOutFormat(), graphs, m2gParams.getLogger(),
104 m2gParams.getRandomizer());
105 } else {
106 DenoptimIO.writeGraphToFile(new File(m2gParams.getOutFile()),
107 m2gParams.getOutFormat(), graphs.get(0), m2gParams.getLogger(),
108 m2gParams.getRandomizer());
109 }
110 }
